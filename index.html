    function mount(){

      // Layout: 4 cards top row, 2 cards bottom-left, how tile bottom-right spanning 2 cols on desktop

      // Accordion behavior: only one open

      grid.querySelectorAll(".card[data-id] .svcBtn").forEach(btn=>{

        btn.addEventListener("click", ()=>{

          const card = btn.closest(".card");

          const open = card.getAttribute("data-open")==="1";


          grid.querySelectorAll(".card[data-id]").forEach(c=>{

            c.setAttribute("data-open","0");

            c.querySelector(".svcBtn").setAttribute("aria-expanded","false");

            const p = c.querySelector(".panel");

            p.style.maxHeight = "0px";

            p.setAttribute("aria-hidden","true");

          });


          if(!open){

            card.setAttribute("data-open","1");

            btn.setAttribute("aria-expanded","true");

            const panel = card.querySelector(".panel");

            panel.setAttribute("aria-hidden","false");

            panel.style.maxHeight = panel.scrollHeight + "px";

          }

        });

      });


      window.addEventListener("resize", ()=>{

        // keep height correct on resize

        grid.querySelectorAll('.card[data-id][data-open="1"] .panel').forEach(p=>{

        });

      });

    }

    function money(n){ return "$" + Math.round(n).toLocaleString(); }


    function calcPainting(){

      const wallSF = Number(document.getElementById("wallSF").value||0);

      const ceilSF = Number(document.getElementById("ceilSF").value||0);

      const coats = Number(document.getElementById("coats").value||1);

      const baseLF = Number(document.getElementById("baseLFp").value||0);


      const wallRate = coats===1 ? 2.25 : 3.00;

      const ceilRate = 2.50;

      const baseRate = 4.50;


      let total = wallSF*wallRate + ceilSF*ceilRate + baseLF*baseRate;

      const min = 500;

      if(total < min) total = min;

      return total;

    }


    function calcFlooring(){

      const floorSF = Number(document.getElementById("floorSF").value||0);

      const mat = document.getElementById("mat").value||"lam";

      const baseLF = Number(document.getElementById("baseLFf").value||0);

      const transQty = Number(document.getElementById("transQty").value||0);


      const install = (mat==="lam" ? 4.25 : 5.00);

      const baseRate = 2.50;

      const trans = 45;


      let total = floorSF*install + baseLF*baseRate + transQty*trans;

      const min = 500;

      if(total < min) total = min;

      return total;

    }


    calcBtn.addEventListener("click", ()=>{

      if(!v) return;

      let total = 0;

      if(v==="painting") total = calcPainting();

      if(v==="flooring") total = calcFlooring();

      amt.textContent = money(total);

      res.classList.add("show");

      res.scrollIntoView({behavior:"smooth", block:"nearest"});

    });
